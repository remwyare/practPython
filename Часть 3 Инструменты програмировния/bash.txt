echo _параметр_ вывод
Python3 	запустить интерпритатор python
history 	список последних команд
pwd  		вывести текущий(рабочий) каталог
cd _путь_  	переместится в другой каталог(путь абсолютный (/) или относительный)
ls  		просмотр каталога
mkdir _назв_	создать каталог, назв без пробелов
rmdir _назв_	удалить каталог
touch _назв_	создание нового файла (.назв - файл скрыт)
export x=100	создание переменной окружения(существует только в окне bash), чтобы существовала всегда поместить в .profile
whoami		вывести пользователя операционной системы
$PATH		каталоги для поиска команд bash
cat _назв_	прочитать содержимое файла
cp readme Desktop/	копирование файла
grep str f1	поиск по регуляерному вырожению





Примеры:
cd ~		переместится в домашний каталог
cd ..		перейти на каталог выше
ls -author	(флаг author, дополнительный вывод владельца ф)
ls -a		(флаг a, вывод скрытых файлов)
ls | less	перенаправление вывода ls в less(программа)
echo $x		вывести переменную кружения
sudo echo Привет от суперпользователя!

grep:
-o		вывести только совпадения, а не всю строку
-i		без учета регистра
grep ^<текст> zen.txt	совпадение в начале строки
grep <текст>.$ zen.txt	поиск совпадений в конце строки
grep [[:digit]] zen.txt поиск цифп
grep ту* zen.txt	т ту туу тууу туууу ... y 0 или больше раз
grep _.*_	. - любой символ __ _а_ _б_ ... _ывфцвфвф_
